CREATE DATABASE ECOMMERCE_DB
GO
USE ECOMMERCE_DB
GO
CREATE TABLE CUSTOMER(
	ID_CUSTOMER INT,
	EMAIL VARCHAR(100),
	NOMBRE VARCHAR(100),
	APELLIDO VARCHAR(100),
	SEXO VARCHAR(100),
	DIRECCION VARCHAR(100),
	FNACIMIENTO VARCHAR(100),
	TELEFONO VARCHAR(100)
)
GO

CREATE TABLE ITEM(
	ID_ITEM INT,
	PRODUCTO VARCHAR(100),
	ID_CATEGORIA INT
)
GO

CREATE TABLE CATEGORY(
	ID_CATEGORIA INT,
	NOMBRE VARCHAR(100),
	DESCRIPCION VARCHAR(100),
	RUTA VARCHAR(100)
)
GO

CREATE TABLE ORDEN(
	ID_ORDEN INT,
	ID_CUSTOMER  INT,
	ID_ITEM INT,
	PRECIO INT,
	ID_VENDEDOR INT,
	FVENTA DATETIME
)
GO

CREATE TABLE ESTADO_ITEM(
	ID_ITEM INT,
	PRODUCTO VARCHAR(100),
	PRECIO INT,
	ESTADO VARCHAR(100)
)
GO
